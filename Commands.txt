# Write the final FAT12 image to a virtual disk (on Windows): http://www.chrysocome.net/dd
./dd.exe if=d:\github\sqlpassion\osdev\main64\kaos64.img od=d:\HDD-flat.vmdk

# Write the final FAT12 image to a virtual disk (on Mac OS):
dd if=kaos64.img of=HDD.vmdk conv=notrunc

# Write the final FAT12 image to a physical disk (where /dev/disk2 is the attached physical hard disk!):
dd if=kaos64.img of=/dev/disk2
